/* Style de base des placeholders */
input::-webkit-input-placeholder { color: #999; }
input:-ms-input-placeholder      { color: #999; }
input:-moz-placeholder           { color: #999; }

/* Au focus, couleur: #fff */
#ph-2:focus::-webkit-input-placeholder { color: #fff; }
#ph-2:focus:-moz-placeholder           { color: #fff; }

/* Au focus, couleur: transparent */
